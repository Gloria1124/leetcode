367.Valid perfect square
========================
Code
----
```javascript
/**
 * @param {number} num
 * @return {boolean}
 */
var isPerfectSquare = function(num) {
    // for(i=1;i<=num;i+=2){
    //    num-=i;
    // }
    // return num===0;
    var l=1,r=num;
    while(l<=r){
        var mid=Math.floor((r+l)/2);
        var sq=mid*mid;
        if(sq==num)return true;
        else if(sq>num)r=mid-1;
        else
            l=mid+1;
    }
    return false;
};
```
