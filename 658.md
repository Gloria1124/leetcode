658.find K closest elements
===========================
Code
----
```javascript
/**
 * @param {number[]} arr
 * @param {number} k
 * @param {number} x
 * @return {number[]}
 */
var findClosestElements = function(arr, k, x) {
    while(arr.length>k){
        var l=0;
        var r=arr.length-1;
        if(Math.abs(arr[l]-x)>Math.abs(arr[r]-x)){
            arr.splice(l,1);
        }
        else if(Math.abs(arr[l]-x)<Math.abs(arr[r]-x)){
            arr.splice(r,1);
        } 
        else{
            if(arr[l]<arr[r])
                arr.splice(r,1);
            else
                arr.splice(l,1);
        }
    }
    return arr;
};
```
