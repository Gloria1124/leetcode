300.Longest increasing subsequence
==================================
Code
----
```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var lengthOfLIS = function(nums) {
    let res=new Array(nums.length).fill(1);
    var x=0;
    for(i=0;i<nums.length;i++){
        for(j=0;j<=i;j++){
            if(nums[j]<nums[i]){
                res[i]=Math.max(res[i],res[j]+1);
               
            }
             x=Math.max(x,res[i]);
        }
    }
    // var x=0;
    // for(let i of res){
    //    x=Math.max(x,i);
    // }
    return x;
};
```
