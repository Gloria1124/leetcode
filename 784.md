784. Letter Case Permutation
============================
Code
----
```javascript
/**
 * @param {string} S
 * @return {string[]}
 */
var letterCasePermutation = function (S) {
    let res = [S];
    S = S.split("");
    var dfs = idx => {
        for (let i = idx; i < S.length; i++) {
            //if S[i] is Number then continue;
            if (S[i] == +S[i]) continue;
            dfs(i + 1);
            S[i] = S[i] == S[i].toLowerCase() ? S[i].toUpperCase() : S[i].toLowerCase();;
            res.push(S.join(""));
        }
    }
    dfs(0);
    return res;
};
```
