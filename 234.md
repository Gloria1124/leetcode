234.Palindrome Linked List
==========================
Code
----
```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {boolean}
 */
var isPalindrome = function (head) {
    let p1 = '';
    let p2 = '';
    while (head != null) {
        p1 = p1 + head.val;
        p2 = head.val + p2;
        head = head.next;
    }
    return p1 === p2;
};
```
