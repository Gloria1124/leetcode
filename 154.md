154.find minimum in rotated sorted array
========================================
Code
----
```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var findMin = function(nums) {
    // let x=nums[0];
    // for (let i of nums){
    //     if(i<=x)x=i;
    // }
    // return x;
    var low=0,high=nums.length-1;
    while(low<high){
        let mid=Math.floor(low+(high-low)/2);
        if(nums[mid]<nums[high]){
            high=mid;
        }
        else if(nums[mid]>nums[high]){
            low=mid+1;
        }
        else
            high-=1
    }
    return nums[low];
};
```
