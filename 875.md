/**
 * @param {number[]} piles
 * @param {number} H
 * @return {number}
 */
var minEatingSpeed = function (piles, H) {
    let l = 1;
    let r = Math.max(...piles);
    while (l < r) {
        let mid = Math.floor(l + (r - l) / 2);

        let h = time(piles, mid);
        if (h > H) {
            l = mid + 1;
        } else {
            r = mid;
        }
    }
    return l;
};

function time(piles, per) {
    let hours = 0;
    for (p of piles) {
        hours += Math.ceil(p / per);
    }
    return hours;
}
