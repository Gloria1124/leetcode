378.Kth smallest element in a sorted matrix
===========================================
Code
----
```javascript
/**
 * @param {number[][]} 
 * @param {number} k
 * @return {number}
 */
var kthSmallest = function(matrix, k) {
    var res=[];
    for(i of matrix){
     res=res.concat(i);
    }
    res.sort((a,b)=>a-b);
    return res[k-1];
};
```
