153.Find Minimum in Rotated Sorted Array
========================================
Code
----
```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var findMin = function(nums) {
 var r=nums[0];
   for(let i of nums){
       if(i<r){
           r=i;
           break;
       }
   }
    return r;
};
```
